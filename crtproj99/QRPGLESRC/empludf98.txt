  CREATE OR REPLACE FUNCTION CUSTDTL(CUID DEC(5,0))
    RETURNS TABLE ( CUSTNM  CHAR(30),
                    CUPHONE CHAR(15),
                    CUMAIL  CHAR(50),
                    CULINE1 CHAR(50),
                    CUZIP   CHAR(10),
                    CUCITY  CHAR(30),
                    CUCOUN  CHAR(2))
    LANGUAGE SQL
    READS SQL DATA
    NO EXTERNAL ACTION
    DETERMINISTIC
    DISALLOW PARALLEL
    RETURN
        SELECT CUSTNM,CUPHONE,CUMAIL,CULINE1,CUZIP,CUCITY,CUCOUN
          FROM PKUMAR4/CUSTOMER
          WHERE CUSTOMER.CUID =CUSTDTL.CUID
